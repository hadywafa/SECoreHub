    public static void MoveZeroes(int[] nums)
    {
        //[0,1,0,3,12]
        //[0,0,1]
        //-------------------------------------
        for (int i = 0; i < nums.Length; i++)
        {
            if (nums[i] == 0)
                continue;

            for (int j = i; j > 0; j--)
            {
                var current = nums[j];
                var prev = nums[j - 1];

                if (prev != 0)
                    break;

                nums[j] = prev;
                nums[j - 1] = current;
            }
        }

        System.Console.WriteLine(nums.HwToString());
    }

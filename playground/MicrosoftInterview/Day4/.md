## üìå 1. [Best Time to Buy and Sell Stock](https://leetcode.com/problems/best-time-to-buy-and-sell-stock/description/)

### ‚ÅâÔ∏è Problem

**Example 1:**

**Input**: `prices = [7,1,5,3,6,4]`  
**Output**: `5`  
**Explanation**: Buy on day 2 (price = 1) and sell on day 5 (price = 6), profit = 6-1 = 5.
Note that buying on day 2 and selling on day 1 is not allowed because you must buy before you sell.

### ‚ö†Ô∏è Solution üòé (Time Limit Exceeded)

```cs
    public static int MaxProfit(int[] prices)
    {
        //[7,1,5,3,6,4]

        int maxProfit = 0;
        for (int i = 0; i < prices.Length; i++)
            for (int j = i; j < prices.Length; j++)
            {
                if (j + 1 == prices.Length)
                    break;
                int diff = prices[j + 1] - prices[i];
                if (diff > maxProfit)
                {
                    maxProfit = diff;
                }
            }
        return maxProfit;
    }
```

### ‚úÖ Solution ü§ñ

```cs
    public static int MaxProfit(int[] prices)
    {
        //[7,1,5,3,6,4]

        int minPrice = prices[0];
        int maxProfit = 0;
        foreach (var price in prices)
        {
            if (price < minPrice)
                minPrice = price;

            if ((price - minPrice) > maxProfit)
                maxProfit = price - minPrice;
        }

        return maxProfit;
    }
```

```cs
    {
        //[7,1,5,3,6,4]

        int minPrice = int.MaxValue, maxProfit = 0;
        foreach (int price in prices)
        {
            minPrice = Math.Min(minPrice, price);
            maxProfit = Math.Max(maxProfit, price - minPrice);
        }
    }
```

## üìå 2. [Valid Parentheses](https://leetcode.com/problems/valid-parentheses/description/)

### ‚ÅâÔ∏è Problem

**Example 1:**

**Input**: `s = "([])"`  
**Output**: `true`

### ‚úÖ Solution üßë‚Äçüíª

```cs
    public static bool IsValidParentheses(string s) {
        Stack<char> stack = new Stack<char>();

        foreach(char c in s){
            if(c=='(' || c=='{' || c=='['){
                stack.Push(c);
            }else{
                if(stack.Count==0)
                    return false;

                char top = stack.Pop();
                if ((c == ')' && top != '(') || (c == '}' && top != '{') || (c == ']' && top != '[')){
                    return false;
                }
            }
        }
        return stack.Count == 0;
    }
```

### ‚úÖ Solution ü§ñ

```cs
    public static bool IsValidParentheses(string s)
    {
        //"([])"
        var dict = new Dictionary<char, char>
        {
            { ')', '(' },
            { ']', '[' },
            { '}', '{' },
        };

        var stack = new Stack<char>();

        foreach (var c in s)
        {
            if (dict.ContainsKey(c)) // closing bracket
            {
                if (stack.Count == 0 || stack.Peek() != dict[c])
                    return false;

                stack.Pop(); // matched open-close pair
            }
            else
            {
                stack.Push(c); // opening bracket
            }
        }

        return stack.Count == 0;
    }
```

## üìå 3. [Assign Cookies](https://leetcode.com/problems/assign-cookies/description/)

### ‚ÅâÔ∏è Problem

**Example 1:**

**Input**: `g = [1,2,3]`  
**Output**: `s = [1,1]`

### ‚ö†Ô∏è Solution üòé (Time Limit Exceeded)

```cs
    public static int FindContentChildren(int[] g, int[] s)
    {
        // int[] g = [1,2,3];
        // int[] s = [1,1];
        //-------------------------
        int happyChildrenCount = 0;
        var cookies = s.Order().ToList();
        // x foreach children
        // if s.contains(x[i]) => happyChildren++; && s.remove(x[i])

        for (int i = 0; i < g.Length; i++)
        {
            var validCookie = cookies.Where(x => g[i] <= x).FirstOrDefault();
            if (validCookie != 0 && cookies.Count > 0)
            {
                happyChildrenCount++;
                var targetDeletedIndex = cookies.IndexOf(validCookie);
                if (targetDeletedIndex != -1)
                    cookies.RemoveAt(targetDeletedIndex);
            }
        }

        return happyChildrenCount;
    }
```

### ‚úÖ Solution üòé

```cs
    public static int FindContentChildren(int[] g, int[] s)
    {
        // int[] g = [1,2,3];
        // int[] s = [1,1];
        //-------------------------
        int happyChildrenCount = 0;
        var cookies = s.Order().ToList();

        for (int i = 0; i < g.Length; i++)
        {
            var currentChildGreed = g[i];

            for (int j = 0; j < cookies.Count; j++)
            {
                var cookieSize = cookies[j];
                if (currentChildGreed <= cookieSize)
                {
                    happyChildrenCount++;
                    cookies.Remove(cookieSize);
                    break;
                }
            }
        }

        return happyChildrenCount;
    }
```

```cs
    public static int FindContentChildren(int[] g, int[] s)
    {
        // int[] g = [1,2,3];
        // int[] s = [1,1];
        //-------------------------
        Array.Sort(g);
        Array.Sort(s);
        //-------------------------
        int happyChildrenCount = 0;
        int pointerForCurrentChild = 0;
        int pointerForUsedCookie = 0;
        //-------------------------
        while (pointerForCurrentChild < g.Length && pointerForUsedCookie < s.Length)
        {
            // var currentChildGreed = g[pointerForCurrentChild];
            // var currentCookieSize = s[pointerForUsedCookie];

            if (s[pointerForUsedCookie] >= g[pointerForCurrentChild])
            {
                happyChildrenCount++;
                pointerForCurrentChild++;
                pointerForUsedCookie++;
            }
            else
            {
                pointerForUsedCookie++;
            }
        }
        return happyChildrenCount;
    }
```

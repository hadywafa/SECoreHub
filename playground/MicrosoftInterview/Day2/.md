```cs
    public static bool IsAnagram(string s, string t)
    {
        if (s.Length != t.Length)
            return false;
        var leftStr = s.ToLower().Select(x => x).Order();
        var rightStr = t.ToLower().Select(x => x).Order();

        return string.Join("", leftStr) == string.Join("", rightStr);
    }
```
